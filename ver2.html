    <script>

        document.addEventListener('DOMContentLoaded', () => {

            

            // Weryfikacja preferencji redukcji ruchu użytkownika

            const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;


            // 1. SEKWENCJA INTRO (Płynne wejście sekcji Hero przy załadowaniu)

            const runHeroIntro = () => {

                if (prefersReducedMotion) return;


                if (typeof gsap !== 'undefined') {

                    const tl = gsap.timeline({ defaults: { ease: 'power3.out' } });

                    

                    tl.from('.main-header', { opacity: 0, y: -20, duration: 0.8 })

                      .from('.hero-title', { opacity: 0, y: 30, duration: 0.8 }, '-=0.4')

                      .from('.hero-subtitle', { opacity: 0, y: 20, duration: 0.6 }, '-=0.5')

                      .from('.meta-item', { opacity: 0, y: 15, duration: 0.5, stagger: 0.1 }, '-=0.4')

                      .from('.hero-actions .btn', { opacity: 0, scale: 0.95, duration: 0.5, stagger: 0.15 }, '-=0.3')

                      .from('.hero-visual', { opacity: 0, scale: 0.9, duration: 1 }, '-=0.6');

                } else {

                    // Natywny fallback, gdyby CDN nie odpowiedział natychmiast

                    document.querySelectorAll('.hero-title, .hero-subtitle, .hero-actions').forEach(el => {

                        el.style.opacity = '1';

                    });

                }

            };


            // 2. PARALAKSA MYSZOWA DLA AMBIENT GLOW (Sygnaturowy Element)

            const initAmbientParallax = () => {

                if (prefersReducedMotion) return;


                const heroGlow = document.getElementById('hero-glow');

                const secondaryGlow = document.getElementById('secondary-glow');


                if (!heroGlow) return;


                window.addEventListener('mousemove', (e) => {

                    const moveX = (e.clientX - window.innerWidth / 2) * 0.03;

                    const moveY = (e.clientY - window.innerHeight / 2) * 0.03;


                    // Wykorzystanie transformacji sprzętowej w celu eliminacji skoków wydajności

                    heroGlow.style.transform = `translate(${moveX}px, ${moveY}px)`;

                    secondaryGlow.style.transform = `translate(${-moveX * 1.5}px, ${-moveY * 1.5}px)`;

                });

            };


            // 3. ŚLEDZENIE POZYCJI KURSORA NA KARTACH (Efekt podświetlenia siatki)

            const initCardMicroInteractions = () => {

                if (prefersReducedMotion) return;


                const cards = document.querySelectorAll('.interactive-card');

                

                cards.forEach(card => {

                    const glow = card.querySelector('.card-glow');

                    

                    card.addEventListener('mousemove', (e) => {

                        const rect = card.getBoundingClientRect();

                        const x = e.clientX - rect.left;

                        const y = e.clientY - rect.top;

                        

                        glow.style.left = `${x}px`;

                        glow.style.top = `${y}px`;

                    });

                });

            };


            // 4. SCROLL-TRIGGERED REVEAL ( Intersection Observer dla sekcji)

            const initScrollReveal = () => {

                if (prefersReducedMotion) return;


                const revealElements = document.querySelectorAll('.scroll-reveal');

                

                const observerOptions = {

                    root: null,

                    threshold: 0.12, // Wyzwolenie animacji, gdy 12% wysokości elementu jest w rzucie ekranu

                    border: '0px'

                };


                const observer = new IntersectionObserver((entries) => {

                    entries.forEach(entry => {

                        if (entry.isIntersecting) {

                            entry.target.classList.add('active');

                            observer.unobserve(entry.target); // Animacja odpala się tylko raz

                        }

                    });

                }, observerOptions);


                revealElements.forEach(el => observer.observe(el));

            };


            // Inicjalizacja wszystkich modułów UI

            runHeroIntro();

            initAmbientParallax();

            initCardMicroInteractions();

            initScrollReveal();

        });
